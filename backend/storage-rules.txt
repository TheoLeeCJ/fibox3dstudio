rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Default: deny all access
    match /{allPaths=**} {
      allow read, write: if false;
    }
    
    // fiboRenders folder: publicly readable (legacy), write only from backend
    match /fiboRenders/{imageId} {
      allow read: if true;
      allow list: if false;
      allow write: if false; // Only backend via service account
    }
    
    // User project files: only accessible by the owning user
    match /users/{userId}/projects/{projectId}/{allPaths=**} {
      // Users can read and write their own project files
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // User renders: publicly readable (for sharing), write only from backend
    match /users/{userId}/renders/{imageId} {
      allow read: if true;  // Public read for generated images
      allow list: if false; // Not listable
      allow write: if false; // Only backend via service account
    }
  }
}
